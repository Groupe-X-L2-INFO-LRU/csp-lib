16static bool sudoku_preassigned_checker(const CSPConstraint *constraint, const size_t *values,
17 const void *data) {
18 // If no grid data is provided, we don't have any constraints to enforce
19 if (!data) {
20 return true;

22
23 // The grid is passed as data parameter
24 const int *initial_grid = (const int *)data;

32 return true;

36 // Note: puzzle digits are 1-9, solver values are 0-8

53 char line[SUDOKU_SIZE * 2]; // Extra space for newlines, etc.
54
55 // Initialize grid to all 0s (blank cells)
56 // Using a safer alternative to memset for buffer initialization
57 for (int i = 0; i < SUDOKU_CELLS; i++) {
58 initial_grid[i] = 0;
59 }
60
61 // Read 9 rows
62 for (size_t row = 0; row < SUDOKU_SIZE; row++) {
63 if (!fgets(line, sizeof(line), in)) {
64 return false; // Failed to read line
65 }
66
67 // Process each character in the line
68 for (size_t col = 0; col < SUDOKU_SIZE; col++) {
69 char c = line[col];
70 if (c >= '1' && c <= '9') {
71 // Convert digit character to integer (1..9)
72 initial_grid[row * SUDOKU_SIZE + col] = c - '0';
73 } else if (c == '.' || c == '0') {
74 // Blank cell - already initialized to 0
75 continue;
76 } else if (!isspace(c)) {
77 // Invalid character
78 return false;
79 }
80 }
81 }
82
83 return true;

111CSPProblem *create_sudoku_problem(const int initial_grid[SUDOKU_CELLS]) {
112 if (!initial_grid) return NULL;
113
114 // Compute the total number of constraints for a Sudoku grid:
115 // 1) Unary constraints for pre-assigned cells: SUDOKU_CELLS (81)
116 // 2) Binary constraints (not-equal) for rows, columns, and 3x3 blocks:
117 // Each unit (row, column, block) has SUDOKU_SIZE*(SUDOKU_SIZE-1)/2 pairs
118 // There are 3*SUDOKU_SIZE units (9 rows, 9 columns, 9 blocks)
119 size_t pairs_per_unit = (SUDOKU_SIZE * (SUDOKU_SIZE - 1)) / 2; // 36 pairs per unit
120 size_t total_binary = pairs_per_unit * (3 * SUDOKU_SIZE); // 36 * 27 = 972 binary constraints
121 size_t total_constraints = total_binary + SUDOKU_CELLS; // 972 + 81 = 1053 constraints
122
123 // Create CSP problem with 81 variables and all constraints
124 CSPProblem *problem = csp_problem_create(SUDOKU_CELLS, total_constraints);
125 if (!problem) return NULL;
126
127 // Set domains for all variables (cells): values 0..8 (representing digits 1..9)
128 for (size_t var = 0; var < SUDOKU_CELLS; var++) {
129 csp_problem_set_domain(problem, var, SUDOKU_SIZE);
130 }
131
132 size_t constraint_index = 0;
133
134 // 1) Create unary constraints for pre-assigned values
135 for (size_t var = 0; var < SUDOKU_CELLS; var++) {
136 CSPConstraint *constraint = csp_constraint_create(1, sudoku_preassigned_checker);
137 if (!constraint) {
138 // We no longer need to manually free each constraint
139 // since we've modified csp_problem_destroy to handle that

145 csp_problem_set_constraint(problem, constraint_index++, constraint);

203 }

206 for (size_t i = 0; i < SUDOKU_SIZE - 1; i++) {
207 for (size_t j = i + 1; j < SUDOKU_SIZE; j++) {
208 CSPConstraint *constraint = csp_constraint_create(2, sudoku_not_equal_checker);
