42size_t select_unassigned_var(const CSPForwardCheckContext *ctx) {
43 // Initialize with maximum values to ensure any variable will be better
44 size_t best = SIZE_MAX; // Tracks the best variable found so far
45 size_t best_count = SIZE_MAX; // Tracks the domain size of the best variable
46
47 // Iterate through all variables in the CSP
48 for (size_t i = 0; i < ctx->num_domains; ++i) {
49 // Skip variables that have already been assigned
50 if (ctx->assigned[i]) continue;
51
52 // Count how many values are still available in this variable's domain
53 size_t count = 0;
54 for (size_t v = 0; v < ctx->original_domain_sizes[i]; ++v) {
55 // If this value is still valid in the domain, increment the counter
56 if (ctx->current_domains[i][v]) ++count;
57 }
58
59 // If this variable has fewer remaining values than the current best,
60 // update our choice (this is the core of the MRV heuristic)
61 if (count < best_count) {
62 best_count = count; // Update the smallest domain size found
63 best = i; // Remember this variable as our current best choice
64 }
65 }
66
67 // Return the variable with the fewest remaining values (most constrained)
68 return best;
69}

98void order_values_lcv(const CSPProblem *csp, CSPForwardCheckContext *ctx, const size_t *values,
99 const void *data, size_t var, size_t *ordered, size_t *n_vals) {
100 // Get the domain size for the variable we're considering
101 size_t dsize = ctx->original_domain_sizes[var];
102
103 // Define a structure to track each value and its conflict count
104 typedef struct {
105 size_t val; // The actual value from the domain
106 size_t conflicts; // How many conflicts this value causes in other variables
107 } pair_t;
108
109 // Allocate array to store all potential values and their conflict counts
110 pair_t *pairs = malloc(dsize * sizeof(pair_t));
111
112 // Counter for the number of valid values we find
113 size_t n = 0;
114
115 // Get the total number of constraints in the CSP
116 size_t num_constraints = csp_problem_get_num_constraints(csp);
117
118 // Cast away const for temporary assignments (we'll modify values temporarily)
119 size_t *vals = (size_t *)values;
120
121 // For each possible value in this variable's domain
122 for (size_t val = 0; val < dsize; ++val) {
123 // Skip values that have been pruned from the domain
124 if (!ctx->current_domains[var][val]) continue;
125
126 // Store this value in our pairs array
127 pairs[n].val = val;
128
129 // Initialize conflict counter for this value
130 size_t conflicts = 0;
131
132 // Temporarily assign this value to the variable
133 vals[var] = val;
134 // For each constraint in the CSP
135 for (size_t ci = 0; ci < num_constraints; ++ci) {
136 // Get the constraint
137 CSPConstraint *con = csp_problem_get_constraint(csp, ci);
138
139 // We only consider binary constraints (between 2 variables)
140 if (csp_constraint_get_arity(con) != 2) continue;
141
142 // Get the two variables involved in this constraint
143 size_t a = csp_constraint_get_variable(con, 0);
144 size_t b = csp_constraint_get_variable(con, 1);
145
146 // Determine which variable in the constraint is the "other" variable
147 // (not the one we're currently assigning)
148 size_t other = SIZE_MAX; // Initialize to an impossible value
149 if (a == var)
150 other = b;
151 else if (b == var)
152 other = a;
153
154 // Skip if this constraint doesn't involve our variable or if the other
155 // variable is already assigned (we only care about future conflicts)
156 if (other == SIZE_MAX || ctx->assigned[other]) continue;
157
158 // For each value in the domain of the other variable
159 for (size_t o_val = 0; o_val < ctx->original_domain_sizes[other]; ++o_val) {
160 // Skip values that have been pruned from the other variable's domain
161 if (!ctx->current_domains[other][o_val]) continue;
162
163 // Temporarily assign this value to the other variable
164 vals[other] = o_val;
165
166 // Check if this assignment violates the constraint
167 // If it does, increment the conflict counter
168 if (!con->check(con, vals, data)) {
169 ++conflicts; // This value for our var conflicts with a value in other var
170 }
171 }
172 }
173 // Store the total conflict count for this value
174 pairs[n].conflicts = conflicts;
175
176 // Move to the next value
177 ++n;
178 }
179
180 // Sort values by conflict count using insertion sort
181 // (values with fewer conflicts should come first - these are less constraining)
182 for (size_t i = 1; i < n; ++i) {
183 // Store the current element we need to insert in the right position
184 pair_t key = pairs[i];
185 size_t j = i;
186
187 // Move elements that have more conflicts than key one position ahead
188 while (j > 0 && pairs[j - 1].conflicts > key.conflicts) {
189 pairs[j] = pairs[j - 1];
190 --j;
191 }
192
193 // Place the key element in its correct position
194 pairs[j] = key;
195 }
196
197 // Copy the ordered values to the output array
198 for (size_t i = 0; i < n; ++i) ordered[i] = pairs[i].val;
199
200 // Set the number of valid values found
201 *n_vals = n;
202
203 // Free the pairs array we allocated
204 free(pairs);
205}

234void prune_neighbors(const CSPProblem *csp, const size_t *values, const void *data,
235 CSPForwardCheckContext *ctx, size_t var, size_t *counts, size_t **pruned) {
236 // Get the total number of constraints in the CSP
237 size_t num_constraints = csp_problem_get_num_constraints(csp);
238
239 // Cast away const for temporary assignments
240 size_t *vals = (size_t *)values;
241
242 // For each constraint in the CSP
243 for (size_t ci = 0; ci < num_constraints; ++ci) {
244 // Get the constraint
245 CSPConstraint *con = csp_problem_get_constraint(csp, ci);
246
247 // We only consider binary constraints (between 2 variables)
248 if (csp_constraint_get_arity(con) != 2) continue;
249
250 // Get the two variables involved in this constraint
251 size_t a = csp_constraint_get_variable(con, 0);
252 size_t b = csp_constraint_get_variable(con, 1);
253
254 // Find the other variable in this constraint (not the one we just assigned)
255 // Using a conditional expression for conciseness
256 size_t other = (a == var ? b : (b == var ? a : SIZE_MAX));
257
258 // Skip if this constraint doesn't involve our variable or if the other
259 // variable is already assigned (no need to prune already assigned vars)
260 if (other == SIZE_MAX || ctx->assigned[other]) continue;
261
262 // Allocate memory to track which values will be pruned from this variable
263 // (only if we haven't already allocated it)
264 if (!pruned[other]) {
265 pruned[other] = malloc(ctx->original_domain_sizes[other] * sizeof(size_t));
266 }
267 // The variable 'var' already has its current value in vals[var]
268
269 // For each value in the domain of the other variable
270 for (size_t o_val = 0; o_val < ctx->original_domain_sizes[other]; ++o_val) {
271 // Skip values that have already been pruned
272 if (!ctx->current_domains[other][o_val]) continue;
273
274 // Temporarily assign this value to the other variable
275 vals[other] = o_val;
276
277 // Check if this combination violates the constraint
278 CSPChecker *check_func = csp_constraint_get_check(con);
279 if (!check_func(con, vals, data)) {
280 // This value is incompatible with our current assignment,
281 // so prune it from the domain
282 ctx->current_domains[other][o_val] = false;
283
284 // Record this pruned value so we can restore it later if needed
285 pruned[other][counts[other]++] = o_val;
286 }
287 }
288 }
289}

311void restore_pruned(CSPForwardCheckContext *ctx, const size_t *counts, size_t **pruned) {
312 // Iterate through all variables in the CSP
313 for (size_t i = 0; i < ctx->num_domains; ++i) {
314 // For each value that was pruned from this variable
315 for (size_t j = 0; j < counts[i]; ++j) {
316 // Restore the value to the domain (mark it as available again)
317 ctx->current_domains[i][pruned[i][j]] = true;
318 }
319
320 // Free the memory used to track pruned values for this variable
321 // (only if it was allocated - the check is implicit since free(NULL) is safe)
322 free(pruned[i]);
323 }
