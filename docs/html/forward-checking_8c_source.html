21static bool fc_is_consistent(const CSPProblem *csp, const size_t *values, const void *data,
22 const CSPForwardCheckContext *ctx) {
23 size_t ncons = csp_problem_get_num_constraints(csp);
24 for (size_t ci = 0; ci < ncons; ++ci) {
25 CSPConstraint *con = csp_problem_get_constraint(csp, ci);
26 size_t ar = csp_constraint_get_arity(con);
27 bool ready = true;
28 // Check if all vars in this constraint have been assigned
29 for (size_t i = 0; i < ar; ++i) {
30 size_t var = csp_constraint_get_variable(con, i);
31 if (!ctx->assigned[var]) {
32 ready = false;
33 break;
34 }
35 }
36 if (!ready) continue;
37 // All variables assigned -> test constraint
38 CSPChecker *check_func = csp_constraint_get_check(con);
39 if (!check_func(con, values, data)) return false;
40 }
41 return true;
42}

81static bool backtrack_fc(const CSPProblem *csp, size_t *values, const void *data,
82 CSPForwardCheckContext *ctx) {
83 assert(csp_initialised());
84
85 // First check if we've received a timeout signal
86 if (timeout_occurred) {
87 return false; // Exit early if timeout occurred
88 }
89
90 // Check if we've assigned all variables - if so, we've found a complete solution
91 for (size_t v = 0; v < ctx->num_domains; ++v) {
92 if (!ctx->assigned[v]) goto proceed; // At least one variable is still unassigned
93 }
94 return true; // All variables are assigned, solution found!
95
96proceed:;
97 // Step 1: Select the next variable to assign using the MRV heuristic
98 // (chooses the variable with the fewest remaining values in its domain)
99 size_t var = select_unassigned_var(ctx);
100
101 // Step 2: Determine the maximum number of values in this variable's domain
102 size_t max_vals = ctx->original_domain_sizes[var];
103
104 // Allocate space for the ordered values buffer
105 size_t *order = malloc(max_vals * sizeof(size_t));
106 size_t n_vals = 0; // Number of values after ordering
107
108 // Step 3: Order the values for this variable using the LCV heuristic
109 // (tries least constraining values first)
110 order_values_lcv(csp, ctx, values, data, var, order, &n_vals);
111
112 // Step 4: Try each possible value for this variable in the order determined by LCV
113 for (size_t i = 0; i < n_vals; ++i) {
114 // Get the current value to try
115 size_t val = order[i];
116
117 // Assign this value to the variable
118 values[var] = val;
119
120 // Mark this variable as assigned for forward checking
121 ctx->assigned[var] = true;
122
123 // Check consistency among already assigned variables
124 if (!fc_is_consistent(csp, values, data, ctx)) {
125 ctx->assigned[var] = false;
126 continue;
127 }
128
129 // No need to check global consistency here since fc_is_consistent already
130 // checks constraints among assigned variables only
131

133 // that would conflict with ou
134 // Step 5: Forward checking - prune values from neighboring variables
135 // that would conflict with our current assignment
136
137 // Allocate arrays for tracking pruned values:
138 // - counts[v] will track how many values were pruned from variable v
139 // - pruned[v] will store which values were pruned from variable v
140 size_t *counts = calloc(ctx->num_domains, sizeof(size_t));
141 size_t **pruned = calloc(ctx->num_domains, sizeof(size_t *));
142
143 // Handle memory allocation failure
144 if (!counts || !pruned) {
145 free(counts);
146 free(pruned);
147 free(order); // Free order memory too
148 ctx->assigned[var] = false;
149 return false;
150 }
151 // Perform the actual pruning of inconsistent values from future variables
152 prune_neighbors(csp, values, data, ctx, var, counts, pruned);
153
154 // Step 6: Recursively try to assign values to the remaining variables
155 if (backtrack_fc(csp, values, data, ctx)) {
156 // Solution found! Clean up and return success
157 free(order);
158 restore_pruned(ctx, counts, pruned);
159 free(counts);
160 free(pruned);
161 return true;
162 }
163
164 // Step 7: If we reach here, this assignment didn't lead to a solution
165 // Backtrack: restore the pruned values and try another value
166 restore_pruned(ctx, counts, pruned);
167 free(counts);
168 free(pruned);
169
170 // Mark this variable as unassigned again
171 ctx->assigned[var] = false;
172 }
173
174 // If we've tried all values and none worked, backtrack further
175 free(order);
176 return false;
177}

194CSPForwardCheckContext *csp_forward_check_context_create(const CSPProblem *csp) {
195 assert(csp_initialised());
196
197 // Get the number of variables in the CSP problem
198 size_t n = csp_problem_get_num_domains(csp);
199
200 // Step 1: Allocate memory for the context structure
201 CSPForwardCheckContext *ctx = malloc(sizeof(*ctx));
202 if (!ctx) return NULL; // Memory allocation failed
203
204 // Store the number of variables
205 ctx->num_domains = n;
206
207 // Step 2: Allocate memory for tracking:
208 // - Which variables have been assigned (initialized to false/0 using calloc)
209 ctx->assigned = calloc(n, sizeof(bool));
210
211 // - The original size of each variable's domain
212 ctx->original_domain_sizes = malloc(n * sizeof(size_t));
213
214 // - The current state of each domain (which values are still available)
215 ctx->current_domains = malloc(n * sizeof(bool *));
216
217 // Check if all memory allocations succeeded
218 if (!ctx->assigned || !ctx->original_domain_sizes || !ctx->current_domains) {
219 // Clean up on failure
220 free(ctx->assigned);
221 free(ctx->original_domain_sizes);
222 free(ctx->current_domains);
223 free(ctx);
224 return NULL;
225 }
226
227 for (size_t i = 0; i < n; ++i) {
228 ctx->original_domain_sizes[i] = csp_problem_get_domain(csp, i);
229 size_t d = ctx->original_domain_sizes[i];
230 ctx->current_domains[i] = calloc(d, sizeof(bool));
231 if (!ctx->current_domains[i]) {
232 // nettoyage en cas d’échec
233 for (size_t j = 0; j < i; ++j) {
234 free(ctx->current_domains[j]);
235 }
236 free(ctx->assigned);
237 free(ctx->original_domain_sizes);
238 free(ctx->current_domains);
239 free(ctx);
240 return NULL;
241 }
242 for (size_t v = 0; v < d; ++v) {
243 ctx->current_domains[i][v] = true;
244 }
245 }
246 // ---- Initial pruning for unary (pre-assigned) constraints ----
247 // Iterate all constraints and prune domains for arity-1 constraints
248 size_t total_cons = csp_problem_get_num_constraints(csp);
249 // Temporary assignment array for testing
250 size_t *tmp_vals = calloc(ctx->num_domains, sizeof(size_t));
251 for (size_t ci = 0; ci < total_cons; ++ci) {
252 CSPConstraint *con = csp_problem_get_constraint(csp, ci);
253 if (csp_constraint_get_arity(con) != 1) continue; // only unary
254 size_t var = csp_constraint_get_variable(con, 0);
255 size_t dsize = ctx->original_domain_sizes[var];
256 CSPChecker *checker = csp_constraint_get_check(con);
257 // Test each value in the variable's domain
258 for (size_t d = 0; d < dsize; ++d) {
259 if (!ctx->current_domains[var][d]) continue;
260 tmp_vals[var] = d;
261 // If the constraint fails, mark this value unavailable
262 if (!checker(con, tmp_vals, NULL)) {
263 ctx->current_domains[var][d] = false;
264 }
265 }
266 // If exactly one value remains, consider variable assigned
267 size_t count = 0;
268 for (size_t d = 0; d < dsize; ++d) {
269 if (ctx->current_domains[var][d]) ++count;
270 }
271 if (count == 1) ctx->assigned[var] = true;
272 }
273 free(tmp_vals);
274 return ctx;
275}

290void csp_forward_check_context_destroy(CSPForwardCheckContext *ctx) {
291 assert(csp_initialised());
292
293 // Safety check - if context is NULL, nothing to free
294 if (!ctx) return;
295
296 // Step 1: Free each variable's domain availability array
297 for (size_t i = 0; i < ctx->num_domains; ++i) {
298 free(ctx->current_domains[i]);
299 }
300
301 // Step 2: Free all the main arrays in the context
302 free(ctx->current_domains); // Array of pointers to domain arrays
303 free(ctx->original_domain_sizes); // Array of original domain sizes
304 free(ctx->assigned); // Array tracking assigned variables
305
306 // Step 3: Free the context structure itself
307 free(ctx);
308}

328bool csp_problem_solve_forward_checking(const CSPProblem *csp, size_t *values, const void *data) {
329 assert(csp_initialised());
330
331 // Step 1: Create a context for forward checking algorithm
332 // This will track domain pruning and variable assignments
333 CSPForwardCheckContext *ctx = csp_forward_check_context_create(csp);
334
335 // Handle memory allocation failure
336 if (!ctx) return false;
337
338 // Step 2: Call the recursive backtracking function with forward checking
339 // This is where the actual solving happens
340 bool solved = backtrack_fc(csp, values, data, ctx);
341
342 // Step 3: Clean up resources regardless of whether a solution was found
343 csp_forward_check_context_destroy(ctx);
344
345 // Return whether we found a solution or not
346 return solved;
347}
