71CSPConstraint *csp_constraint_create(size_t arity, CSPChecker *check) {
72 assert(csp_initialised());
73 assert(arity > 0); // Constraints must involve at least one variable
74 assert(check != NULL); // Must provide a valid constraint checking function
75 assert(printf("Creating constraint with arity %lu\n", arity));
76
77 // Allocate memory for the constraint structure with a flexible array member
78 // We allocate exactly the right amount of space for the 'variables' array
79 // based on the constraint's arity (number of variables involved)
80 CSPConstraint *constraint = malloc(sizeof(CSPConstraint) + arity * sizeof(size_t));
81
82 if (constraint != NULL) {
83 // Initialize the constraint properties
84 constraint->arity = arity;
85 constraint->check = check;
86
87 // Initialize all variables to 0
88 // (These will be set to actual variable indices later with csp_constraint_set_variable)
89 // Using a safer alternative to memset for buffer initialization
90 for (size_t i = 0; i < arity; i++) {
91 constraint->variables[i] = 0;
92 }
93 }
94
95 // Return the created constraint, or NULL if allocation failed
96 return constraint;
97}

127bool csp_constraint_to_check(const CSPConstraint *constraint, size_t index) {
128 assert(csp_initialised());
129
130 // Check if all variables in this constraint have indices less than 'index'
131 // This determines if the constraint can be checked at the current stage of solving
132 //
133 // In backtracking, we only check constraints when all their variables have been assigned
134 // Since we assign variables in order (0, 1, 2, ...), if any variable in the constraint
135 // has index >= the current assignment index, it means that variable hasn't been assigned yet
136 for (size_t i = 0; i < constraint->arity; i++) {
137 if (constraint->variables[i] >= index) {
138 // At least one variable in this constraint hasn't been assigned yet,
139 // so we can't check this constraint at this point
140 return false;
141 }
142 }
143
144 // All variables in this constraint have been assigned,
145 // so we can check if the constraint is satisfied
146 return true;
147}

149CSPProblem *csp_problem_create(size_t num_domains, size_t num_constraints) {
150 assert(csp_initialised());
151 assert(num_domains > 0); // A CSP must have at least one variable
152 // We allow 0 constraints for testing purposes (though a real CSP would have constraints)
153
154 assert(printf("Creating CSP problem with %lu domains and %lu constraints\n", num_domains,
155 num_constraints));
156
157 // Step 1: Allocate memory for the main CSP problem structure
158 CSPProblem *csp = malloc(sizeof(CSPProblem));
159 if (csp != NULL) {
160 // Step 2: Allocate memory for the domains array
161 // Each entry will store the size of the domain for that variable
162 // Using calloc to initialize all domains to 0
163 csp->domains = calloc(num_domains, sizeof(size_t));
164
165 if (csp->domains != NULL) {
166 // Step 3: Allocate memory for the constraints array
167 // This is an array of pointers to constraint objects
168 // Only allocate if we have constraints, otherwise set to NULL
169 csp->constraints =
170 num_constraints > 0 ? malloc(num_constraints * sizeof(CSPConstraint *)) : NULL;
171
172 // This condition handles both the case where num_constraints is 0 (no allocation
173 // needed) and where the allocation succeeded
174 if (num_constraints == 0 || csp->constraints != NULL) {
175 // Initialize all constraint pointers to NULL
176 // (They'll be set later with csp_problem_set_constraint)
177 for (size_t i = 0; i < num_constraints; i++) {
178 csp->constraints[i] = NULL;
179 }
180
181 // Store the problem dimensions
182 csp->num_domains = num_domains;
183 csp->num_constraints = num_constraints;
184 } else {
185 // Failed to allocate constraints array - clean up and return NULL
186 free(csp->domains);
187 free(csp);
188 csp = NULL;
189 }
190 } else {
191 // Failed to allocate domains array - clean up and return NULL
192 free(csp);
193 csp = NULL;
194 }
195 } else {
196 // Failed to allocate CSP problem structure - just return NULL
197 csp = NULL;
198 }
199
200 // Return the created CSP problem, or NULL if any allocation failed
201 return csp;
202}

205 assert(csp_initialised());

207 csp->num_constraints));
208
209 // Free all constraints before freeing the constraints array

211 if (csp->constraints[i] != NULL) {

214 }
215

217 free(csp->domains);

266bool csp_problem_is_consistent(const CSPProblem *csp, const size_t *values, const void *data,
267 size_t index) {
268 assert(csp_initialised());
269
270 // Iterate through all constraints in the CSP problem
271 for (size_t i = 0; i < csp->num_constraints; i++) {
272 CSPConstraint *constraint = csp->constraints[i];
273
274 // Two conditions must be met to check a constraint:
275 // 1. All variables involved in the constraint must already be assigned
276 // (checked by csp_constraint_to_check)
277 // 2. The constraint must be satisfied with the current assignment
278 // (checked by constraint->check)
279
280 // If the constraint should be checked and it's not satisfied, the
281 // current assignment is inconsistent
282 if (csp_constraint_to_check(constraint, index) &&
283 !constraint->check(constraint, values, data)) {
284 return false; // This assignment violates a constraint
285 }
286 }
287
288 // If we reach here, all applicable constraints are satisfied
289 return true;
290}

297bool csp_problem_backtrack(const CSPProblem *csp, size_t *values, const void *data, size_t index) {
298 assert(csp_initialised());
299
300 // Base case: If we've assigned values to all variables, we've found a solution
301 if (index == csp->num_domains) {
302 return true;
303 }
304
305 // Try each possible value in the domain of the current variable
306 for (size_t i = 0; i < csp->domains[index]; i++) {
307 // Step 1: Assign this value to the current variable
308 values[index] = i;
309
310 // Step 2: Check if this assignment is consistent with all constraints
311 // that involve only variables we've assigned so far
312 if (csp_problem_is_consistent(csp, values, data, index + 1)) {
313 // Step 3: If consistent, recursively try to assign the next variable
314 if (csp_problem_backtrack(csp, values, data, index + 1)) {
315 // Solution found!
316 return true;
317 }
318
319 // If we reach here, this path didn't lead to a solution
320 // We'll implicitly try the next value (no need to undo assignment)
321 }
322 }
323
324 // If we've tried all values and none worked, backtrack to the previous variable
325 return false;
326}
