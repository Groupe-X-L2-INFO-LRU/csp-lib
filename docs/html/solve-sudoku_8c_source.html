1// filepath: solve-sudoku.c
2//
3// This program models the Sudoku problem as a Constraint Satisfaction Problem (CSP)
4// using the CSP library. It uses forward checking with MRV and LCV heuristics
5// to efficiently solve the puzzle. The Sudoku grid is represented as 81 variables
6// with domains 1..9 and all-different constraints on rows, columns, and 3x3 blocks.

19// Holds the initial puzzle values: 0 for blank, 1..9 for pre-filled cells
20static int initial_grid[SUDOKU_CELLS];

33bool sudoku_preassigned(const CSPConstraint *constraint, const size_t *values, const void *data) {
34 // Determine which variable this constraint applies to
35 size_t var = csp_constraint_get_variable(constraint, 0);
36 int fixed = initial_grid[var];
37 if (fixed == 0) {
38 // No pre-assigned value, always satisfied
39 return true;
40 }
41 // Puzzle digits are 1..9, solver values are 0..8, so compare with fixed-1
42 return values[var] == (size_t)(fixed - 1);
43}

55bool sudoku_not_equal(const CSPConstraint *constraint, const size_t *values, const void *data) {
56 size_t a = csp_constraint_get_variable(constraint, 0);
57 size_t b = csp_constraint_get_variable(constraint, 1);
58 return values[a] != values[b];

82bool read_puzzle(FILE *in) {
83 char line[32];
84 for (size_t i = 0; i < SUDOKU_SIZE; i++) {
85 if (!fgets(line, sizeof(line), in)) return false;
86 for (size_t j = 0; j < SUDOKU_SIZE; j++) {
87 char c = line[j];
88 if (c >= '1' && c <= '9') {
89 initial_grid[i * SUDOKU_SIZE + j] = c - '0';
90 } else {
91 // blank
92 initial_grid[i * SUDOKU_SIZE + j] = 0;
93 }
94 }
95 }
96 return true;
97}

99int main(int argc, char *argv[]) {
100 FILE *in = stdin;
101 bool using_file = false;
102
103 if (argc == 2) {
104 in = fopen(argv[1], "r");
105 if (!in) {
106 // Using snprintf for safer string formatting
107 char error_msg[256];
108 snprintf(error_msg, sizeof(error_msg), "Cannot open file: %s\n", argv[1]);
109 fputs(error_msg, stderr);
110 return EXIT_FAILURE;
111 }
112 using_file = true;
113 }
114
115 // Read the puzzle
116 bool read_success = read_puzzle(in);
117
118 // Close file if we opened one
119 if (using_file) {
120 fclose(in);
121 }
122
123 if (!read_success) {
124 fputs("Failed to read puzzle\n", stderr);
125 return EXIT_FAILURE;
126 }
127
128 // Initialize CSP library
129 csp_init();
130
131 // Pre-allocate an array for the solution values (0..8)
132 // and set initial assignments for pre-filled cells
133 size_t solution[SUDOKU_CELLS] = {0};
134 for (size_t v = 0; v < SUDOKU_CELLS; ++v) {
135 if (initial_grid[v] != 0) {
136 // Fixed value in puzzle: store zero-based index
137 solution[v] = (size_t)(initial_grid[v] - 1);
138 }
139 }
140
141 // Compute the total number of binary constraints for rows, cols, blocks
142 size_t pairs_per_unit = (SUDOKU_SIZE * (SUDOKU_SIZE - 1)) / 2;
143 size_t total_binary = pairs_per_unit * (3 * SUDOKU_SIZE);
144 // Add unary constraints for each cell (pre-assignment)
145 size_t total_constraints = total_binary + SUDOKU_CELLS;
146
147 // Create CSP problem: 81 variables, total_constraints constraints
148 CSPProblem *problem = csp_problem_create(SUDOKU_CELLS, total_constraints);
149
150 // Set domains: each cell can take values 0..8 (i.e. digits 1..9)
151 for (size_t v = 0; v < SUDOKU_CELLS; v++) {
152 csp_problem_set_domain(problem, v, SUDOKU_SIZE);
153 }
154
155 size_t ci = 0;
156 // 1) Unary constraints for pre-assigned cells
157 for (size_t v = 0; v < SUDOKU_CELLS; v++) {
158 CSPConstraint *c = csp_constraint_create(1, (CSPChecker *)sudoku_preassigned);
159 csp_constraint_set_variable(c, 0, v);
160 csp_problem_set_constraint(problem, ci++, c);
161 }
162
163 // 2) All-different constraints for rows
164 for (size_t r = 0; r < SUDOKU_SIZE; r++) {
165 for (size_t a = 0; a < SUDOKU_SIZE - 1; a++) {
166 for (size_t b = a + 1; b < SUDOKU_SIZE; b++) {
167 size_t v1 = r * SUDOKU_SIZE + a;
168 size_t v2 = r * SUDOKU_SIZE + b;
169 CSPConstraint *c = csp_constraint_create(2, (CSPChecker *)sudoku_not_equal);
170 csp_constraint_set_variable(c, 0, v1);
171 csp_constraint_set_variable(c, 1, v2);
172 csp_problem_set_constraint(problem, ci++, c);
173 }
174 }
175 }
176 // 3) All-different constraints for columns
177 for (size_t cidx = 0; cidx < SUDOKU_SIZE; cidx++) {
178 for (size_t a = 0; a < SUDOKU_SIZE - 1; a++) {
179 for (size_t b = a + 1; b < SUDOKU_SIZE; b++) {
180 size_t v1 = a * SUDOKU_SIZE + cidx;
181 size_t v2 = b * SUDOKU_SIZE + cidx;
182 CSPConstraint *c = csp_constraint_create(2, (CSPChecker *)sudoku_not_equal);
183 csp_constraint_set_variable(c, 0, v1);
184 csp_constraint_set_variable(c, 1, v2);
185 csp_problem_set_constraint(problem, ci++, c);
186 }
187 }
188 }
189 // 4) All-different constraints for 3x3 blocks
190 for (size_t block = 0; block < SUDOKU_SIZE; block++) {
191 size_t br = (block / 3) * 3;
192 size_t bc = (block % 3) * 3;
193 // Collect cell indices in this block
194 size_t idxs[9];
195 size_t k = 0;
196 for (size_t dr = 0; dr < 3; dr++) {
197 for (size_t dc = 0; dc < 3; dc++) {
198 idxs[k++] = (br + dr) * SUDOKU_SIZE + (bc + dc);
199 }
200 }
201 // Add pairwise constraints within block
202 for (size_t a = 0; a < 9 - 1; a++) {
203 for (size_t b = a + 1; b < 9; b++) {
204 CSPConstraint *c = csp_constraint_create(2, (CSPChecker *)sudoku_not_equal);
205 csp_constraint_set_variable(c, 0, idxs[a]);
206 csp_constraint_set_variable(c, 1, idxs[b]);
207 csp_problem_set_constraint(problem, ci++, c);
208 }
209 }
210 }
211
212 // Solve the Sudoku with forward checking and heuristics
213 bool solved = csp_problem_solve_forward_checking(problem, solution, NULL);
214
215 // Cleanup: no need to manually destroy constraints - csp_problem_destroy will handle that
216 csp_problem_destroy(problem);
217
218 // Print result
219 if (solved) {
220 print_sudoku(solution);
221 } else {
222 printf("No solution found\n");
223 }
224
225 csp_finish();
226 return EXIT_SUCCESS;
227}
